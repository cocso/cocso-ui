---
title: Tooltip
description: A floating label that provides supplementary information when an element is hovered or focused.
---

<Section>
  <Section.Title>Overview</Section.Title>
  <Section.Text>
    Tooltip is a compound component built on Base UI's Tooltip primitive. It renders a short text label when the user hovers or focuses a trigger element. Wrapping multiple tooltips with a shared <code>Tooltip.Provider</code> lets you coordinate delay and open behavior across them.
  </Section.Text>
  <Section.UnorderedList>
    <Section.ListItem>Composed with a compound API: <code>Tooltip.Provider</code>, <code>Tooltip</code>, <code>Tooltip.Trigger</code>, <code>Tooltip.Content</code>, <code>Tooltip.Arrow</code></Section.ListItem>
    <Section.ListItem>Wrapping a group of tooltips with <code>Tooltip.Provider</code> shares hover delay settings.</Section.ListItem>
    <Section.ListItem>Supports <code>side</code> and <code>align</code> positioning props on <code>Tooltip.Content</code>.</Section.ListItem>
    <Section.ListItem><code>Tooltip.Arrow</code> renders an arrow caret pointing toward the trigger.</Section.ListItem>
  </Section.UnorderedList>
</Section>

<Section>
  <Section.Title>Default</Section.Title>
  <Section.Text>A tooltip that appears when hovering the trigger button.</Section.Text>
  <ComponentExample name="tooltip-default">
```tsx
"use client";

import { Button, Tooltip } from "@cocso-ui/react";

export default function TooltipDefault() {
  return (
    <div className="p-4">
      <Tooltip.Provider>
        <Tooltip>
          <Tooltip.Trigger render={<Button variant="secondary" />}>
            Hover me
          </Tooltip.Trigger>
          <Tooltip.Content>
            This is a tooltip
            <Tooltip.Arrow />
          </Tooltip.Content>
        </Tooltip>
      </Tooltip.Provider>
    </div>
  );
}
```
  </ComponentExample>
</Section>

<Section>
  <Section.Title>Positions</Section.Title>
  <Section.Text>Use the <code>side</code> prop on <code>Tooltip.Content</code> to control which side of the trigger the tooltip appears on.</Section.Text>
  <ComponentExample name="tooltip-positions">
```tsx
"use client";

import { Button, Tooltip } from "@cocso-ui/react";

export default function TooltipPositions() {
  return (
    <div className="flex flex-wrap gap-3 p-4">
      <Tooltip.Provider>
        {(["top", "bottom", "left", "right"] as const).map((side) => (
          <Tooltip key={side}>
            <Tooltip.Trigger render={<Button variant="outline" />}>{side}</Tooltip.Trigger>
            <Tooltip.Content side={side}>
              Tooltip on {side}
              <Tooltip.Arrow />
            </Tooltip.Content>
          </Tooltip>
        ))}
      </Tooltip.Provider>
    </div>
  );
}
```
  </ComponentExample>
</Section>
