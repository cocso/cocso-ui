<Section>
  <Section.Title>Overview</Section.Title>
  <Section.Text>
    The Select component wraps the native HTML select element with consistent styling, label support, and validation states. Use it when a user needs to choose a single value from a predefined list.
  </Section.Text>
</Section>

<Section>
  <Section.Title>Default</Section.Title>
  <Section.Text>
    A basic select with a label and options.
  </Section.Text>
  <ComponentExample name="select-default">
    ```tsx
    import { Select } from "@cocso-ui/react";

    export default function SelectDefault() {
      return (
        <div className="flex w-full max-w-sm flex-col gap-4 p-4">
          <Select label="Country">
            <option value="">Select a country</option>
            <option value="kr">South Korea</option>
            <option value="us">United States</option>
            <option value="jp">Japan</option>
          </Select>
        </div>
      );
    }
    ```
  </ComponentExample>
</Section>

<Section>
  <Section.Title>Size</Section.Title>
  <Section.Text>
    Four sizes are available to match surrounding content density.
  </Section.Text>
  <ComponentExample name="select-sizes">
    ```tsx
    import { Select } from "@cocso-ui/react";

    export default function SelectSizes() {
      return (
        <div className="flex w-full max-w-sm flex-col gap-4 p-4">
          <Select label="Large" size="large"><option>Option</option></Select>
          <Select label="Medium" size="medium"><option>Option</option></Select>
          <Select label="Small" size="small"><option>Option</option></Select>
          <Select label="X-Small" size="x-small"><option>Option</option></Select>
        </div>
      );
    }
    ```
  </ComponentExample>
</Section>

<Section>
  <Section.Title>States</Section.Title>
  <Section.Text>
    Use <code>description</code> for help text, <code>error</code> for validation messages, and <code>disabled</code> to prevent interaction.
  </Section.Text>
  <ComponentExample name="select-states">
    ```tsx
    import { Select } from "@cocso-ui/react";

    export default function SelectStates() {
      return (
        <div className="flex w-full max-w-sm flex-col gap-4 p-4">
          <Select description="Choose your preferred option." label="With description">
            <option>Option A</option>
            <option>Option B</option>
          </Select>
          <Select error="Please select an option." label="With error">
            <option value="">Select...</option>
          </Select>
          <Select disabled label="Disabled">
            <option>Cannot change</option>
          </Select>
        </div>
      );
    }
    ```
  </ComponentExample>
</Section>
