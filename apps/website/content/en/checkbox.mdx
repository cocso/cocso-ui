---
title: Checkbox
description: A control that lets users toggle a boolean or intermediate state.
---

<Section>
  <Section.Title>Overview</Section.Title>
  <Section.Text>
    The Checkbox component allows users to select or deselect an option. It supports three states: on, off, and intermediate (indeterminate). Use it in forms, filter panels, and multi-selection scenarios.
  </Section.Text>
  <Section.Text>
    Checkbox is a controlled component — manage its state via `status` and `onChange`. Use the `intermediate` state for "select all" patterns where only some children are selected. For mutually exclusive choices, use RadioGroup instead.
  </Section.Text>
</Section>

<Section>
  <Section.Title>Accessibility</Section.Title>
  <Section.UnorderedList>
    <Section.ListItem>Renders with <code>role="checkbox"</code> and <code>aria-checked</code> states for assistive technologies.</Section.ListItem>
    <Section.ListItem><code>Space</code> toggles the checkbox. <code>Tab</code> moves focus between checkboxes.</Section.ListItem>
    <Section.ListItem>The <code>intermediate</code> state maps to <code>aria-checked="mixed"</code>, communicating a partial selection to assistive technologies.</Section.ListItem>
    <Section.ListItem>When <code>label</code> is provided, it is automatically associated with the checkbox for screen readers.</Section.ListItem>
    <Section.ListItem>When <code>disabled</code>, the checkbox is visually dimmed and excluded from keyboard navigation.</Section.ListItem>
  </Section.UnorderedList>
</Section>

<Section>
  <Section.Title>Import</Section.Title>
  ```tsx
  import { Checkbox } from "@cocso-ui/react";
  ```
</Section>

<Section>
  <Section.Title>Default</Section.Title>
  <Section.Text>
    A basic checkbox controlled with local state.
  </Section.Text>
  <ComponentExample name="checkbox-default">
    ```tsx
    "use client";

    import { Checkbox } from "@cocso-ui/react";
    import { useState } from "react";

    export default function CheckboxDefault() {
      const [status, setStatus] = useState<"on" | "off" | "intermediate">("off");

      return (
        <div className="flex flex-col gap-4 p-4">
          <Checkbox label="Accept terms and conditions" onChange={setStatus} status={status} />
        </div>
      );
    }
    ```
  </ComponentExample>
</Section>

<Section>
  <Section.Title>States</Section.Title>
  <Section.Text>
    The checkbox supports <code>on</code>, <code>off</code>, and <code>intermediate</code> states. Use <code>disabled</code> to prevent interaction.
  </Section.Text>
  <ComponentExample name="checkbox-states">
    ```tsx
    "use client";

    import { Checkbox } from "@cocso-ui/react";

    export default function CheckboxStates() {
      return (
        <div className="flex flex-col gap-4 p-4">
          <Checkbox label="Checked" onChange={() => {}} status="on" />
          <Checkbox label="Unchecked" onChange={() => {}} status="off" />
          <Checkbox label="Intermediate" onChange={() => {}} status="intermediate" />
          <Checkbox disabled label="Disabled" onChange={() => {}} status="on" />
        </div>
      );
    }
    ```
  </ComponentExample>
</Section>
