---
title: Month Picker
description: A date picker that opens a calendar popover for selecting a month and year.
---

<Section>
  <Section.Title>Overview</Section.Title>
  <Section.Text>MonthPicker renders a trigger element that opens a Dropdown containing a month-year grid. It is a controlled component; pass value and onValueChange to manage the selected month. Navigation arrows move by year rather than by month. The calendar is localized to English by default.</Section.Text>
</Section>

<Section>
  <Section.Title>Default</Section.Title>
  <Section.Text>Control the selected month with the value and onValueChange props. The returned Date represents the first day of the selected month.</Section.Text>
  <ComponentExample name="month-picker-default">
    ```tsx
    "use client";

    import { MonthPicker } from "@cocso-ui/react";
    import { useState } from "react";

    export default function MonthPickerDefault() {
      const [value, setValue] = useState<Date | null>(null);

      return (
        <div className="p-4">
          <MonthPicker onValueChange={setValue} value={value ?? undefined} />
          {value && <p className="mt-2 text-sm text-neutral-500">Selected: {value.toLocaleDateString("en-US", { year: "numeric", month: "long" })}</p>}
        </div>
      );
    }
    ```
  </ComponentExample>
</Section>
