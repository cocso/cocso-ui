---
title: Day Picker
description: A date picker that opens a calendar popover for selecting a single date.
---

<Section>
  <Section.Title>Overview</Section.Title>
  <Section.Text>DayPicker renders a trigger element that opens a Dropdown containing an inline calendar. It is a controlled component; pass value and onValueChange to manage the selected date. The calendar is localized to English by default.</Section.Text>
  <Section.UnorderedList>
    <Section.ListItem>Use DayPicker for selecting a specific calendar date.</Section.ListItem>
    <Section.ListItem>Use <code>minDate</code> and <code>maxDate</code> to constrain the selectable range.</Section.ListItem>
    <Section.ListItem>For month-level selection, use MonthPicker instead.</Section.ListItem>
    <Section.ListItem>Provide a custom <code>trigger</code> element for better visual integration with form layouts.</Section.ListItem>
  </Section.UnorderedList>
</Section>

<Section>
  <Section.Title>Accessibility</Section.Title>
  <Section.UnorderedList>
    <Section.ListItem>The calendar grid supports arrow key navigation between days.</Section.ListItem>
    <Section.ListItem><code>Escape</code> closes the dropdown and returns focus to the trigger.</Section.ListItem>
    <Section.ListItem>Dates outside the <code>minDate</code>–<code>maxDate</code> range are marked with <code>aria-disabled</code>.</Section.ListItem>
    <Section.ListItem>Month navigation buttons allow moving forward and backward through months.</Section.ListItem>
  </Section.UnorderedList>
</Section>

<Section>
  <Section.Title>Import</Section.Title>
  ```tsx
  import { DayPicker } from "@cocso-ui/react";
  ```
</Section>

<Section>
  <Section.Title>Default</Section.Title>
  <Section.Text>Control the selected date with the value and onValueChange props. When no custom trigger is provided, it renders as a default button.</Section.Text>
  <ComponentExample name="day-picker-default">
    ```tsx
    "use client";

    import { Button, DayPicker } from "@cocso-ui/react";
    import { CalendarMonthIcon } from "@cocso-ui/react-icons";
    import { useState } from "react";

    export default function DayPickerDefault() {
      const [value, setValue] = useState<Date | null>(null);

      return (
        <div className="p-4">
          <DayPicker
            onValueChange={setValue}
            trigger={
              <Button prefix={<CalendarMonthIcon size={16} />} variant="outline">
                {value ? value.toLocaleDateString("ko-KR") : "Select date"}
              </Button>
            }
            value={value ?? undefined}
          />
        </div>
      );
    }
    ```
  </ComponentExample>
</Section>

    { name: "disabled", type: "boolean", description: "Disables the trigger button and prevents the calendar from opening." },
