---
title: Switch
description: A toggle control for binary on/off settings.
---

<Section>
  <Section.Title>Overview</Section.Title>
  <Section.Text>
    The Switch component provides a visual toggle for activating or deactivating a setting. Built on a headless accessibility foundation, it supports multiple sizes, semantic color variants, and label position configuration.
  </Section.Text>
</Section>

<Section>
  <Section.Title>Default</Section.Title>
  <Section.Text>
    A basic switch controlled with local state.
  </Section.Text>
  <ComponentExample name="switch-default">
    ```tsx
    "use client";

    import { Switch } from "@cocso-ui/react";
    import { useState } from "react";

    export default function SwitchDefault() {
      const [checked, setChecked] = useState(false);

      return (
        <div className="flex flex-col gap-4 p-4">
          <Switch checked={checked} label="Enable notifications" onCheckedChange={setChecked} />
        </div>
      );
    }
    ```
  </ComponentExample>
</Section>

<Section>
  <Section.Title>Size</Section.Title>
  <Section.Text>
    Three sizes are available. Choose based on surrounding layout density.
  </Section.Text>
  <ComponentExample name="switch-sizes">
    ```tsx
    import { Switch } from "@cocso-ui/react";

    export default function SwitchSizes() {
      return (
        <div className="flex flex-col gap-4 p-4">
          <Switch defaultChecked label="Large" size="large" />
          <Switch defaultChecked label="Medium" size="medium" />
          <Switch defaultChecked label="Small" size="small" />
        </div>
      );
    }
    ```
  </ComponentExample>
</Section>

<Section>
  <Section.Title>Variant</Section.Title>
  <Section.Text>
    Color variants convey the semantic meaning of the toggle. Use <code>primary</code> for general settings and state variants for actions with specific outcomes.
  </Section.Text>
  <ComponentExample name="switch-variants">
    ```tsx
    import { Switch } from "@cocso-ui/react";

    export default function SwitchVariants() {
      return (
        <div className="flex flex-col gap-4 p-4">
          <Switch defaultChecked label="Primary" variant="primary" />
          <Switch defaultChecked label="Success" variant="success" />
          <Switch defaultChecked label="Error" variant="error" />
          <Switch defaultChecked label="Warning" variant="warning" />
          <Switch defaultChecked label="Info" variant="info" />
        </div>
      );
    }
    ```
  </ComponentExample>
</Section>

<Section>
  <Section.Title>Label Position</Section.Title>
  <Section.Text>
    Use the <code>position</code> prop to control whether the label appears to the left or right of the switch. The default is <code>right</code>.
  </Section.Text>
</Section>

<Section>
  <Section.Title>Disabled</Section.Title>
  <Section.Text>
    Set <code>disabled</code> to prevent the user from toggling the switch.
  </Section.Text>
</Section>
