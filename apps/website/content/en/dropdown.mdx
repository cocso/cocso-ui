title: Dropdown

<Section>
  <Section.Title>Overview</Section.Title>
  <Section.Text>
    Dropdown is a compound component built on Base UI's Menu primitive. It renders a floating list of actions anchored to a trigger element. Use it for contextual actions such as edit, duplicate, and delete, or for a list of commands associated with a specific item.
  </Section.Text>
  <Section.UnorderedList>
    <Section.ListItem>Composed with a compound API: <code>Dropdown</code>, <code>Dropdown.Trigger</code>, <code>Dropdown.Content</code>, <code>Dropdown.Item</code></Section.ListItem>
    <Section.ListItem>Keyboard navigation with arrow keys; <code>Escape</code> closes the menu.</Section.ListItem>
    <Section.ListItem>Items support a <code>prefix</code> slot for a leading icon.</Section.ListItem>
  </Section.UnorderedList>
</Section>

<Section>
  <Section.Title>Default</Section.Title>
  <Section.Text>A dropdown menu triggered by an icon button, containing labeled action items.</Section.Text>
  <ComponentExample name="dropdown-default">
```tsx
"use client";

import { Button, Dropdown } from "@cocso-ui/react";
import { ContentCopyIcon, DeleteIcon, MoreHorizIcon, PencilIcon } from "@cocso-ui/react-icons";

export default function DropdownDefault() {
  return (
    <div className="p-4">
      <Dropdown>
        <Dropdown.Trigger render={<Button variant="secondary" svgOnly />}>
          <MoreHorizIcon size={18} />
        </Dropdown.Trigger>
        <Dropdown.Content>
          <Dropdown.Item prefix={<PencilIcon size={16} />}>Edit</Dropdown.Item>
          <Dropdown.Item prefix={<ContentCopyIcon size={16} />}>Duplicate</Dropdown.Item>
          <Dropdown.Item prefix={<DeleteIcon size={16} />}>Delete</Dropdown.Item>
        </Dropdown.Content>
      </Dropdown>
    </div>
  );
}
```
  </ComponentExample>
</Section>

<Section>
  <Section.Title>Prefix Icon</Section.Title>
  <Section.Text>Pass an icon node to the <code>prefix</code> prop of <code>Dropdown.Item</code> to render a leading icon before the label.</Section.Text>
</Section>
