---
title: One-Time Password Field
description: A split input field for entering one-time passwords and verification codes.
---

<Section>
  <Section.Title>Overview</Section.Title>
  <Section.Text>OneTimePasswordField renders a row of individual input slots for entering a one-time password or verification code. Built on input-otp, it supports paste, keyboard navigation, and accessibility out of the box. Use onValueChange to receive the current full value as a string.</Section.Text>
</Section>

<Section>
  <Section.Title>Default</Section.Title>
  <ComponentExample name="otp-default">
    ```tsx
    "use client";

    import { OneTimePasswordField } from "@cocso-ui/react";
    import { useState } from "react";

    export default function OTPDefault() {
      const [value, setValue] = useState("");

      return (
        <div className="p-4">
          <OneTimePasswordField onValueChange={setValue} />
          {value && <p className="mt-2 text-sm text-neutral-500">Value: {value}</p>}
        </div>
      );
    }
    ```
  </ComponentExample>
</Section>
