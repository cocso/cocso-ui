---
title: Dialog
description: A modal dialog that interrupts the user's flow with important content and requires a response.
---

<Section>
  <Section.Title>Overview</Section.Title>
  <Section.Text>
    Dialog is a compound component built on Base UI's Dialog primitive. It renders a modal overlay that traps focus and blocks interaction with the rest of the page until the user dismisses it. Use Dialog for confirmations, forms, or content that requires an explicit user decision.
  </Section.Text>
  <Section.UnorderedList>
    <Section.ListItem>Composed with a compound API: <code>Dialog</code>, <code>Dialog.Trigger</code>, <code>Dialog.Content</code>, <code>Dialog.Title</code>, <code>Dialog.Description</code>, <code>Dialog.Close</code></Section.ListItem>
    <Section.ListItem>Focus is trapped inside while the dialog is open.</Section.ListItem>
    <Section.ListItem>Pressing the <code>Escape</code> key closes the dialog by default.</Section.ListItem>
    <Section.ListItem>Three content sizes are supported: <code>small</code>, <code>medium</code>, <code>large</code>.</Section.ListItem>
  </Section.UnorderedList>
</Section>

<Section>
  <Section.Title>Default</Section.Title>
  <Section.Text>A dialog with a title, description, and action buttons.</Section.Text>
  <ComponentExample name="dialog-default">
```tsx
"use client";

import { Button, Dialog } from "@cocso-ui/react";

export default function DialogDefault() {
  return (
    <div className="p-4">
      <Dialog>
        <Dialog.Trigger render={<Button />}>Open Dialog</Dialog.Trigger>
        <Dialog.Content>
          <Dialog.Title>Confirm Action</Dialog.Title>
          <Dialog.Description>
            Are you sure you want to proceed? This action cannot be undone.
          </Dialog.Description>
          <div className="mt-4 flex justify-end gap-2">
            <Dialog.Close render={<Button variant="secondary" />}>Cancel</Dialog.Close>
            <Dialog.Close render={<Button variant="primary" />}>Confirm</Dialog.Close>
          </div>
        </Dialog.Content>
      </Dialog>
    </div>
  );
}
```
  </ComponentExample>
</Section>

<Section>
  <Section.Title>Size</Section.Title>
  <Section.Text>The <code>size</code> prop on <code>Dialog.Content</code> controls the maximum width of the dialog panel.</Section.Text>
  <ComponentExample name="dialog-sizes">
```tsx
"use client";

import { Button, Dialog } from "@cocso-ui/react";

export default function DialogSizes() {
  return (
    <div className="flex flex-wrap gap-3 p-4">
      {(["small", "medium", "large"] as const).map((size) => (
        <Dialog key={size}>
          <Dialog.Trigger render={<Button variant="secondary" />}>{size}</Dialog.Trigger>
          <Dialog.Content size={size}>
            <Dialog.Title>Dialog — {size}</Dialog.Title>
            <Dialog.Description>This is a {size} dialog.</Dialog.Description>
            <div className="mt-4 flex justify-end">
              <Dialog.Close render={<Button variant="primary" />}>Close</Dialog.Close>
            </div>
          </Dialog.Content>
        </Dialog>
      ))}
    </div>
  );
}
```
  </ComponentExample>
</Section>
